/*
二分答案，一般是检验条件是否合理
    计数性,枚举某个物品 出现/使用 的数量 <=限制
    可行性,某个条件是否能够达到
一般都是，足够大是可以的。我们尝试让答案变小，但是肯定会存在某些条件，使得它不能无限变小
当然，在二分前。我们要想一想答案的最大值以及最小值
*/
#include <iostream>
#include <algorithm>
using namespace std;
const int N  = 100010;
typedef long long LL;
int n,m,k;
int t[N],c[N];
bool check(int x)
{
    LL sum = 0;
    for (int i = 1;i <= n;i ++)
    {
        if (t[i] > x)
        {
            sum += ((LL)t[i] - x) * c[i];
        }
    }
    return sum <= m;
}
int main()
{
    cin >> n >> m >> k;
    int mx = 0;
    for (int i = 1;i <= n;i ++)
    {
        cin >> t[i] >> c[i];
        mx = max(mx, t[i]);
    }
    
    int l = k,r = mx;
    while (l < r)
    {
        int mid = l + r >> 1;
        if (check(mid)) r = mid;
        else l = mid + 1;
    }
    cout << r;
    return 0;
}

脉冲神经网络
2023/5/25

题目链接：https://www.acwing.com/problem/content/4011/

在本题中，你需要实现一个 SNN（spiking neural network，脉冲神经网络）的模拟器。

一个 SNN 由以下几部分组成：

  1. 神经元：按照一定的公式更新内部状态，接受脉冲并可以发放脉冲
  2. 脉冲源：在特定的时间发放脉冲
  3. 突触：连接神经元-神经元或者脉冲源-神经元，负责传递脉冲

神经元会按照一定的规则更新自己的内部状态。

本题中，我们对时间进行离散化处理，即设置一个时间间隔 $\Delta{t}$，仅考虑时间间隔整数倍的时刻 $t=k\Delta{t}(k \in Z^+)$，按照下面的公式，从 $k-1$ 时刻的取值计算 $k$ 时刻的变量的取值：

$$v_k=v_{k-1} + \Delta{t}(0.04v_{k-1}^2 + 5v_{k-1} + 140 - u_{k-1}) + I_k$$
$$u_k=u_{k-1} + \Delta{t}a(bv_{k-1} - u_{k-1})$$

其中 $v$ 和 $u$ 是神经元内部的变量，会随着时间而变化，$a$ 和 $b$ 是常量，不会随着时间变化；其中 $I_k$ 表示该神经元在 $k$ 时刻接受到的所有脉冲输入的强度之和，如果没有接受到脉冲，那么 $I_k=0$。

当进行上面的计算后，如果满足 $v_k \ge 30$，神经元会发放一个脉冲，脉冲经过突触传播到其他神经元；同时，$v_k$ 设为 $c$ 并且 $u_k$ 设为 $u_k + d$，其中 $c$ 和 $d$ 也是常量。

图 $1$ 展示了一个神经元 $v$ 变量随时间变化的曲线。

图 $1$: 神经元 $v$ 变量随时间变化的曲线

突触表示的是神经元-神经元、脉冲源-神经元的连接关系，包含一个入结点和一个出结点（可能出现自环和重边）。

当突触的入结点（神经元或者脉冲源）在 $k$ 时刻发放一个脉冲，那么在传播延迟 $D(D > 0)$ 个时刻以后，也就是在 $k+D$ 时刻突触的出结点（神经元）会接受到一个强度为 $w$ 的脉冲。

脉冲源在每个时刻以一定的概率发放一个脉冲，为了模拟这个过程，每个脉冲源有一个参数 $0 < r \le 32767$，并统一采用以下的伪随机函数：

C++ 版本：

static unsigned long next = 1;


/* RAND_MAX assumed to be 32767 */
int myrand(void) {
    next = next * 1103515245 + 12345;
    return((unsigned)(next/65536) % 32768);
}

Python 版本：

next = 1

    global next
    next = (next * 1103515245 + 12345) % (2 ** 64)
    return (next // 65536) % 32768

Java 版本：

int myrand() {
    next = next * 1103515245 + 12345;
    return (int)((Long.divideUnsigned(next, 65536)) % 32768);
}

在每个时间刻，按照编号顺序从小到大，每个脉冲源调用一次上述的伪随机函数，当 $r > \text{myrand()}$ 时，在当前时间刻发放一次脉冲，并通过突触传播到神经元。

进行仿真的时候，已知 $0$ 时刻各个神经元的状态，从 $1$ 时刻开始按照上述规则进行计算，直到完成 $T$ 时刻的计算，再输出 $T$ 时刻神经元的 $v$ 值和发放的脉冲次数分别的最小值和最大值。

规定输入数据中结点按如下方式顺序编号：$[0, N-1]$ 为神经元的编号，$[N, N+P-1]$ 为脉冲源的编号。

代码中请使用双精度浮点类型。

输入格式

输入的第一行包括四个以空格分隔的正整数 $N$ $S$ $P$ $T$，表示一共有 $N$ 个神经元，$S$ 个突触和 $P$ 个脉冲源，输出时间刻 $T$ 时神经元的 $v$ 值。

输入的第二行是一个正实数 $\Delta{t}$，表示时间间隔。

输入接下来的若干行，每行有以空格分隔的一个正整数 $R_N$ 和六个实数 $v$ $u$ $a$ $b$ $c$ $d$，按顺序每一行对应 $R_N$ 个具有相同初始状态和常量的神经元：其中 $v$ $u$ 表示神经元在时刻 $0$ 时的变量取值；$a$ $b$ $c$ $d$ 为该神经元微分方程里的四个常量。保证所有的 $R_N$ 加起来等于 $N$。它们从前向后按编号顺序描述神经元，每行对应一段连续编号的神经元的信息。

输入接下来的 $P$ 行，每行是一个正整数 $r$，按顺序每一行对应一个脉冲源的 $r$ 参数。

输入接下来的 $S$ 行，每行有以空格分隔的两个整数 $s(0 \le s < N+P)$、$t(0 \le t < N)$ 、一个实数 $w(w \ge 0)$ 和一个正整数 $D$，其中 $s$ 和 $t$ 分别是入结点和出结点的编号；$w$ 和 $D$ 分别表示脉冲强度和传播延迟。

输出格式

输出共有两行，第一行由两个近似保留 $3$ 位小数的实数组成，分别是所有神经元在时刻 $T$ 时变量 $v$ 的取值的最小值和最大值。

第二行由两个整数组成，分别是所有神经元在整个模拟过程中发放脉冲次数的最小值和最大值。

只要按照题目要求正确实现就能通过，不会因为计算精度的问题而得到错误答案。

数据范围

数据共 $3$ 种类型：

  • 类型 $1$ 满足，$1 \le T,N,S,P,D \le 100$。
  • 类型 $2$ 满足，$1 \le T,N,S,P,D \le 1000$。
  • 类型 $3$ 满足，$1 \le T \le 10^5$，$1 \le N,S,P \le 1000$，$1 \le D \le 10$。

#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;
const int N = 5010;
const double INF = 1e8;
static unsigned long _next = 1;

/* RAND_MAX assumed to be 32767 */
int myrand(void) {
    _next = _next * 1103515245 + 12345;
    return((unsigned)(_next/65536) % 32768);
}
int n,s,p,T;
double dt;
int rn;
double v[N],u[N],a[N],b[N],c[N],d[N];
double I[1024][N / 2];//只有脉冲强度有时间跨度的概念，由时间k，更新时间k + D的状态。所以要把I开两维
//剩下的参数只是由i - 1时刻推导到i，这个由最普通的变量的更新就可以模拟。
int r[N],cnt[N];

int e[N],h[N],ne[N],idx;
double W[N];
int D[N];
void add(int a,int b,double w,int d)
{
    e[idx] = b,ne[idx] = h[a],W[idx] = w,D[idx] = d,h[a] = idx ++;
}
int main()
{
    memset(h,-1,sizeof h);
    cin >> n >> s >> p >> T;
    cin >> dt;
    for (int i = 0;i < n;)
    {
       cin >> rn;
       double vv,uu,aa,bb,cc,dd;
       cin >> vv >> uu >> aa >> bb >> cc >> dd;
       for (int j = 0;j < rn;j ++,i ++)
       {
           v[i] = vv, u[i] = uu, a[i] = aa, b[i] = bb, c[i] = cc, d[i] = dd;
       }
    }
    for (int i = n;i < n + p;i ++) cin >> r[i];
    
    int mod = 0;
    while (s --)
    {
        int a,b,dd;
        double ww;
        cin >> a >> b >> ww >> dd;
        add(a,b,ww,dd);
        mod = max(mod, dd + 1);
    }
    
    //接下来开始模拟
    //模拟当中涉及时间
    for (int i = 0;i < T;i ++)//每一时刻
    {
        int t = i % mod;
        for (int j = n;j < n + p;j ++)//检查脉冲源是否有发射信号
        {
            if (r[j] > myrand())
            {
                for (int k = h[j]; ~k;k = ne[k])
                {
                    int ver = e[k];
                    I[(t + D[k]) % mod][ver] += W[k];//通过边，K是一个idx，一个整数，充当“边”，去索引D和W;
                }
            }
        }
        
        for (int j = 0;j < n;j ++)//检查神经元是否有发射信号
        {
            
            //我有个疑惑：我到底是应该先累计I呢？还是应该先更新v[i]和u[i];
            //因为同一个时刻，对于同一个神经元，可能同时它既会接收脉冲，也要在V>=30时发射脉冲。那么这个次序要搞清楚
            //默认顺序是先接收，后发射。因为假设我们写的时候它先发射，那么意味着它已经是V>=30的了，那么意味着它在上一个时刻
            //接收脉冲后，已经是V>=30了，理应在上个时刻就释放。这样就与题目有矛盾了
            
            //而且最关键的一点，题目有原话：
            double vv = v[j], uu = u[j];
            v[j] = vv + dt * (0.04 * vv * vv + 5 * vv + 140 - uu) + I[t][j];
            u[j] = uu + dt * a[j] * (b[j] * vv - uu);
            //题目原话：当进行完上面的计算后，如果满足V >= 30;那么才释放。意味着它一定是先累计，再释放。这也符合人的逻辑
            if (v[j] >= 30)
            {
                for (int k = h[j]; ~k;k = ne[k])
                {
                    int ver = e[k];
                    I[(t + D[k]) % mod][ver] += W[k];
                }
                v[j] = c[j];
                u[j] += d[j];
                cnt[j] ++;//表示神经元向周边发射了一次脉冲
            }
        }
        memset(I[t], 0,sizeof I[t]);//环形数组清空一下
    }
    double minv = INF, maxv = -INF;
    int minc = INF, maxc = -INF;

    for (int i = 0; i < n; i ++ )
    {
        minv = min(minv, v[i]);
        maxv = max(maxv, v[i]);
        minc = min(minc, cnt[i]);
        maxc = max(maxc, cnt[i]);
    }

    printf("%.3lf %.3lf\n", minv, maxv);
    printf("%d %d\n", minc, maxc);

    return 0;
}

矩阵乘法 + 快速幂求解经典斐波那契数列
2023/5/11

题目链接：https://www.acwing.com/solution/content/47987/

#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;
const int N = 3;
typedef long long LL;
int n,m;
void mul(int c[][N],int a[][N],int b[][N])//这里直接是传引用的形式进来
{
    static int temp[N][N];
    memset(temp,0,sizeof temp);
    for (int i = 0;i < N;i ++)
        for (int j = 0;j < N;j ++)
            for (int k = 0;k < N;k ++)
                temp[i][j] = (temp[i][j] + (LL)a[i][k] * b[k][j]) % m;

    memcpy(c,temp,sizeof temp);        
}
int main()
{
    cin >> n >> m;
    /*f1[3][3] 是一个矩阵；写成二维使得我们只需要写一个乘法
    1 1 1
    0 0 0
    0 0 0
    */
    int f1[3][3] = {0,1,0};
    int A[3][3] = {
        {0,1,0},
        {1,1,1},
        {0,0,1}
    };

    for (int i = n;i;i >>= 1)
    {
        if (i & 1) mul(f1,f1,A);//F1 = F1 * A;
        mul(A,A,A);//A = A * A;
    }
    cout << f1[0][2];
    return 0;
}

大模拟乒乓球
2023/5/11

题目链接：https://www.acwing.com/problem/content/description/1505/
因为本题要处理的对象比较多，且关系交织复杂，所以模拟起来就很容易乱。粗略的想一想，本题有两个对象要处理：一共有“球员桌子”两个随机变量，并且桌子还有空与不空一个变量，一共三个随机变量需要处理，根据乘法原理，至少八种情况要讨论(哭o(╥﹏╥)o)。这里看一下朴素的思想要讨论什么，

#include <iostream>
#include <algorithm>
#include <cmath>
#include <queue>
using namespace std;
const int N = 10010,M = 110,INF = 0x3f3f3f3f;
int n,m,k;
struct Person
{
    int arrive_time,play_time;
    int start_time,waiting_time;
    bool operator < (const Person &t) const
    {
        if(start_time == t.start_time) return arrive_time < t.arrive_time;
        else return start_time < t.start_time;
    }
    bool operator > (const Person &t) const
    {
        return arrive_time > t.arrive_time;
    }
};
struct Table
{
    int id;
    int end_time;
    bool operator >(const Table &t) const
    {
        if(end_time == t.end_time) return id > t.id;
        else return end_time > t.end_time;
    }
};
bool is_vip_table[M];
int table_cnt[M];
vector<Person> persons;
void assign(priority_queue<Person,vector<Person>,greater<Person>>& ps,
            priority_queue<Table,vector<Table>,greater<Table>>& ts)//要加&,对原来的堆进行修改 
{
    auto p = ps.top();ps.pop();
    auto t = ts.top();ts.pop();
    p.waiting_time = round((t.end_time - p.arrive_time) / 60.0);//四舍五入，round计算的是浮点数
    p.start_time = t.end_time;
    table_cnt[t.id] ++;
    persons.push_back(p);//因为是按时间从顾客的堆里每次取人的，所以这里自然就是按时间顺序存储的答案。
    ts.push({t.id,t.end_time + p.play_time});
}
string get_time(int secs)
{
    char str[20];
    sprintf(str, "%02d:%02d:%02d", secs / 3600, secs % 3600 / 60, secs % 60);//将秒数重新转化成时、分、秒。
    return str;
}
int main()
{
    //搞四个堆，存储普通球员、VIP球员、普通桌子、VIP桌子
    cin >> n;
    priority_queue<Person,vector<Person>,greater<Person>> normal_persons;
    priority_queue<Person,vector<Person>,greater<Person>> vip_persons;
    //哨兵设置，使得POP堆的时候不需要特判空(很巧妙)
    normal_persons.push({INF});
    vip_persons.push({INF});
    for(int i = 0;i < n;i ++)
    {
        int hour,minute,second;
        int play_time,is_vip;
        scanf("%d:%d:%d %d %d",&hour,&minute,&second,&play_time,&is_vip);
        int s = hour * 3600 + minute * 60 + second;
        play_time = play_time * 60;
        play_time = min(play_time,2 * 3600);
        if(is_vip) vip_persons.push({s,play_time});
        else normal_persons.push({s,play_time});
    }

    priority_queue<Table,vector<Table>,greater<Table>> normal_tables;
    priority_queue<Table,vector<Table>,greater<Table>> vip_tables;
    normal_tables.push({-1,INF});//哨兵设置
    vip_tables.push({-1,INF});
    cin >> k >> m;
    for(int i = 0;i < m;i ++)
    {
        int vip_id;
        cin >> vip_id;
        is_vip_table[vip_id] = true;
    }
    //初始化所有球桌
    for(int i = 1;i <= k;i ++)
    {
        if(is_vip_table[i]) vip_tables.push({i,8 * 3600});
        else normal_tables.push({i,8 * 3600});
    }
    //依次按到达时间去看对于每来的一位球员(在堆中要想实现遍历只能用whlie()来做了，不支持for的枚举)
    while(normal_persons.size() > 1 || vip_persons.size() > 1)
    {
        auto np = normal_persons.top();
        auto vp = vip_persons.top();
        int arrive_time = min(np.arrive_time,vp.arrive_time);
        //处理球台未工作的情况，这里和“银行排队”不同的是我们将“未工作”情况转化成了“工作”的情况
        //好处是我们不需要判断“未工作”时，来的该同学是VIP球员(应该分配VIP球桌)还是普通球员(应该分配普通球桌)
        //通过给小于球员到达时间的球台(那些早结束的)，赋值成球员到达时间
        //这里就是end_time结束早的，统一看成end_time为刚好顾客来的时间(即：刚好不空，两者成功交接)
        while(normal_tables.top().end_time < arrive_time)
        {
            auto t = normal_tables.top();
            normal_tables.pop();
            t.end_time = arrive_time;
            normal_tables.push(t);
        }
        while(vip_tables.top().end_time < arrive_time)
        {
            auto t = vip_tables.top();
            vip_tables.pop();
            t.end_time = arrive_time;
            vip_tables.push(t);
        }
        //对来的每一位顾客；
        //选个球桌
        auto nt = normal_tables.top();
        auto vt = vip_tables.top();
        int end_time = min(nt.end_time,vt.end_time);

        if(end_time >= 21 * 3600) break;
        //下面的分类讨论中，由于VIP很特殊，所以一般判断VIP在前头。这样可以省下些判断
        //如果来的顾客是vip，并且是vip球桌(这个“与”写的很妙，而且y总对“与”下文的else“或”也能很好的讨论)
        if(vp.arrive_time <= end_time && vt.end_time == end_time) assign(vip_persons,vip_tables);
        else if(np.arrive_time < vp.arrive_time)//在顾客不是vip或者没有vip球桌的情况下如果来的是普通球员
        {//对于来的一个球员，下面分两种情况讨论空出的桌子：
           if(nt > vt) assign(normal_persons,vip_tables);//如果迎接普通球员，有vip桌子
           else assign(normal_persons,normal_tables);//如果迎接普通球员，有普通桌子
        }
        else//在顾客不是vip或者没有vip球桌的情况下如果来的是VIP球员
        {//对于来的一个球员，下面分两种情况讨论空出的桌子：
            if(nt > vt) assign(vip_persons,vip_tables);//如果迎接VIP球员，只有VIP桌子
            //其实把这句话删掉也是对的，因为如果有vt，VIP桌子，就一定不会有VIP球员(从路径描述可以看出)
            else assign(vip_persons,normal_tables);//如果迎接VIP球员，只有普通桌子
        }
    }


    sort(persons.begin(),persons.end());

    for(auto person : persons)
    {
        cout << get_time(person.arrive_time) << ' ' << get_time(person.start_time) << ' ';
        cout <<person.waiting_time << endl;
    }
    cout << table_cnt[1];
    for(int i = 2;i <= k;i ++)
    {
        cout << ' ' << table_cnt[i];
    }
    cout << endl;
    return 0;
}

ICS-1-C语言机制
2022/4/14

C程序执行的两个视角

静态: C代码的连续一段 总能对应到 一段连续的机器指令
动态： C代码执行的状态总能对应到 机器的状态
●源代码视角 

◾函数、变量、指针...

●机器指令视角

◾ 寄存器、内存、地址...

两个视角共同之处： 内存
◾ 代码、变量（源代码视角） = 地址 + 长度（机器指令视角）
◾ 内存 = 代码 + 数据 + 堆栈
因此理解C程序执行最重要的就是内存模型

C语言中一切皆是地址
指针都是纯粹的，类型声明只不过是对一段内存的解读方式 

#include <stdio.h>
void printptr(void *p)//空指针是非常纯粹的，它就是一个纯粹的地址，强制类型转换更改了内存的解读方式
{
    printf("*p = %p       *p = %016lx\n",p,*(long *)p);//对于p来说，首先它是一个指针，剩下的那个long代表对指针指向的内存的一个解读
}
int x;
int main(int argc,char *argv[])
{
    printptr(main);//代码
    printptr(&main);
    printptr(&x);//数据
    printptr(&argc);//堆栈
    printptr(argv);
    printptr(&argv);
    printptr(argv[0]);
}

C Type System
类型：对一段内存的解读方式。C里所有的数据都可以理解成是地址（指针）+ 类型（对地址的解读） 

#include <stdio.h>
#include <assert.h>
int main(int argc,char *argv[])
{
    int (*f)(int,char *[]) = main;
    ////利用f可以直接调用函数，刚才上面看过了它能够执行的原因，利用一个指针就可以直接把main的代码读出来
    if(argc != 0)
    {
        char ***a = &argv, *first = argv[0], ch = argv[0][0];
        printf("arg = \"%s\";    ch = '%c'\n",first, ch);//输出结果：arg = "./a.out";    ch = '.'
        assert(***a == ch);
        /*
        assert(***a == ch)等价于
        if(***a == ch)
        {
            
        }
        else
        {
            printf("ERROR\n");
            exit(1);
        }如果程序出现bug能够直接中断
        */
        f(argc - 1,argv + 1);
    }
}

C语言机制使得代码可读并且可维护
2022/4/14


1、软件里有很多隐藏的dependence（依赖关系），有一些额外的，代码中没有体现和约束的“规则”
2、往往如果增加一个需求，就需要改动很多地方
一个例子是如果要模拟一个寄存器的工作，现在要加一个新的寄存器，就需要更改：X定义的位置、X1定义的位置、计算模拟的位置、写回的位置。如果这四个位置分布在一篇代码里还好，如果每个位置都分布在不同的文件中那就麻烦了

● 一处改了，另外一处就忘了(例如数组中加了一个寄存器忘记更新NREG...)
● 减少依赖关系 -> 降低代码耦合程度


好处在于只需要定义一次列表就可以对列表进行不同的处理，这种形式很利于维护
不至于把同一段代码copy很多次，很大程度上促进了代码的复用

#include <stdint.h>
#define NREG 4

typedef uint8_t u8;//定义一个8bit的int
u8 pc = 0,R[NREG];
/*
小知识：
还存在一个intptr_t这种类型，保证了int里面能够容纳指针
对于(int)ptr这种操作是非常危险的，如果在32位机里面ptr是4字节，int也是4字节那还好
但是如果把同样一段代码移植到了64位机里面int是4字节，而ptr是8字节，此时如果将指针cast成int的话那就JJ
*/

int main()
{


    //方法1
    u8 R[] = {
        0,0,0,0,0......init_pc,
    };//这种定义不好维护，很容易就漏写0，而且如果需求有更改（要再多加一个寄存器）比较麻烦


    //方法2
    enum {RA,R1 ..... PC};
    u8 R[] = {
        [RA] = 0,
        [R1] = 0,
        ...
        [PC] = init_pc,
    }//这种定义的手法，给予了代码很好的可读性，使得代码“不言自明”，而且如果要多加一个寄存器的话也只需要改列表，然后多加一个就好了

    return 0;
}
