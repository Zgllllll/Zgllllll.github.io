Dijkstra求最短路
2022/11/19

给定一个 $n$ 个点 $m$ 条边的有向图，图中可能存在重边和自环，所有边权均为正值。

请你求出 $1$ 号点到 $n$ 号点的最短距离，如果无法从 $1$ 号点走到 $n$ 号点，则输出 $-1$。

输入格式

第一行包含整数 $n$ 和 $m$。

接下来 $m$ 行每行包含三个整数 $x,y,z$，表示存在一条从点 $x$ 到点 $y$ 的有向边，边长为 $z$。

输出格式

输出一个整数，表示 $1$ 号点到 $n$ 号点的最短距离。

如果路径不存在，则输出 $-1$。

数据范围

$1 \le n \le 500$,
$1 \le m \le 10^5$,
图中涉及边长均不超过10000。

输入样例：

3 3
1 2 2
2 3 1
1 3 4

输出样例：

3

DFS暴力求最短路。如果加上一个Depth参数，可以限制深度。
DFS求最短路是在遍历到每一个点的时候做更新。而dijkstra求最短路，是在扩展每个点的时候做。两者的流程有些不同
不过核心思想都是一样的，只要一个点被多次更新的时候，遇到更小的，就去更新它。
如题目：https://www.acwing.com/file_system/file/content/whole/index/content/7299482/
在遍历每个点的时候做：

#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;
const int N = 510,M = 100010;
int g[N][N];
int n,m;
int dist[N];
bool st[N];
void dfs(int u,int curlen)
{
    //cout << u << endl;
    if(dist[u] <= curlen) return;
    if(u == n)
    {
        if(dist[n] > curlen) dist[n] = curlen;
    }
    else
    {
        if(dist[u] > curlen) dist[u] = curlen;
        for(int i = 1;i <= n;i ++)
        {
            if(g[u][i] != 0x3f3f3f3f) dfs(i,curlen + g[u][i]);
        }
    }
    return;
}

int main()
{
    memset(dist,0x3f,sizeof dist);
    memset(g,0x3f,sizeof g);
    cin >> n >> m;
    while(m --)
    {
        int a,b,c;
        cin >> a >> b >> c;
        g[a][b] = min(g[a][b],c);
    }
    dfs(1,0);
    if(dist[n] == 0x3f3f3f3f) printf("-1");
    else cout << dist[n];
    return 0;
}

在扩展的时候做：

#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;
const int N = 510;
int n,m;
int g[N][N],d[N];
bool st[N];
int dijkstra()
{
    memset(d,0x3f,sizeof d);
    d[1] = 0;
    for(int i = 0;i < n - 1;i ++)
    {
        int t = -1;
        for(int j = 1;j <= n;j ++)
        {
            if(!st[j] && (t == -1 || d[t] > d[j]))
            {
                t = j;
            }
        }
        st[t] = true;
        for(int j = 1;j <= n;j ++)
        {
            if(d[j] > d[t])
            {
                d[j] = min(d[j],d[t] + g[t][j]);
            }
        }
    }
    if (d[n] == 0x3f3f3f3f) return -1;
    return d[n];
}

int main()
{
    memset(g,0x3f,sizeof g);
    scanf("%d%d",&n,&m);
    while(m --)
    {
        int a,b,w;
        scanf("%d%d%d",&a,&b,&w);
        g[a][b] = min(g[a][b],w);
    }
    int res = dijkstra();
    printf("%d",res);
    return 0;
}

在扩展更新的模式下，为了保证扩展的正确性，都会先把“起点”处理一下。而直接对那个点做dfs，就事先不需要做什么处理，因为我们会在递归的每个点里去做。这是两种流程的特点

微博转发
2022/11/19

题目链接：https://www.acwing.com/problem/content/1564/

微博被称为中文版的 Twitter。

微博上的用户既可能有很多关注者，也可能关注很多其他用户。

因此，形成了一种基于这些关注关系的社交网络。

当用户在微博上发布帖子时，他/她的所有关注者都可以查看并转发他/她的帖子，然后这些人的关注者可以对内容再次转发…

现在给定一个社交网络，假设只考虑 $L$ 层关注者，请你计算某些用户的帖子的最大可能转发量。

补充

如果 $B$ 是 $A$ 的关注者，$C$ 是 $B$ 的关注者，那么 $A$ 的第一层关注者是 $B$，第二层关注者是 $C$。

输入格式

第一行包含两个整数，$N$ 表示用户数量，$L$ 表示需要考虑的关注者的层数。

假设，所有的用户的编号为 $1 \sim N$。

接下来 $N$ 行，每行包含一个用户的关注信息，格式如下：

M[i] user_list[i]

M[i] 是第 $i$ 名用户关注的总人数，user_list[i] 是第 $i$ 名用户关注的 M[i] 个用户的编号列表。

最后一行首先包含一个整数 $K$，表示询问次数，然后包含 $K$ 个用户编号，表示询问这些人的帖子的最大可能转发量。

输出格式

按顺序，每行输出一个被询问人的帖子最大可能转发量。

假设每名用户初次看到帖子时，都会转发帖子，只考虑 $L$ 层关注者。

数据范围

$1 \le N \le 1000$,
$1 \le L \le 6$,
$1 \le M[i] \le 100$,
$1 \le K \le N$

输入样例：

7 3
3 2 3 4
0
2 5 6
2 3 1
2 3 4
1 4
1 5
2 2 6

输出样例：

4
5


这道题就可以深刻体会到，BFS能够直接求边权为1的最短路的好处了。
这道题也要走最短的一条路，怪不得DFS半天过不去，在DFS的过程当中还要维护一个每个点的最短路径
当然，最好直接用BFS做就行了….

#include <iostream>
#include <algorithm>
#include <cstring>
#include <vector>
using namespace std;
const int N = 1010;
vector<int> g[N];
int n,depth;
bool st[N];
int cnt;
int dist[N];
void dfs(int u,int curlen) 
{
    if(curlen >= dist[u]) return;
    if(curlen > depth) return;
    dist[u] = curlen;
    for(auto j : g[u])
    {
        dfs(j,curlen + 1);
    }
    return;
}

int main()
{
    scanf("%d%d",&n,&depth);
    int num,b;
    for(int i = 1;i <= n;i ++)
    {
        scanf("%d",&num);
        while(num --)
        {
            scanf("%d",&b);
            g[b].push_back(i);
        }
    }
    int k,id;
    scanf("%d",&k);
    while(k --)
    {
        scanf("%d",&id);
        cnt = 0;
        //cout << id << endl;
        memset(st,0,sizeof st);
        memset(dist,0x3f,sizeof dist);
        dfs(id,0);
        for(int i = 1;i <= n;i ++)
            if(dist[i] != 0x3f3f3f3f) cnt ++;
        printf("%d\n",cnt - 1);
    }
    return 0;
}

加油站的建造位置必须使加油站与距离它最近的房屋的距离尽可能远。

与此同时，它还必须保证所有房屋都在其服务范围内。

现在，给出了城市地图和加油站的几个候选位置，请你提供最佳建议。

#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;
const int N = 2010,INF = 0x3f3f3f3f;
typedef pair<int,int> PII;
int n,m,k,D;
int g[N][N];
int get_num(string s)
{
    if(s[0] == 'G') return stoi(s.substr(1)) + n;
    return stoi(s);
}
int dist[N];
bool st[N];
bool dijkstra(int S,int& maxD,int& sumD,int& minID)
{
    memset(st,0,sizeof st);
    memset(dist,0x3f,sizeof dist);
    dist[S] = 0;
    for(int i = 1;i <= n + m;i ++)
    {
        int t = -1;
        for(int j = 1;j <= n + m;j ++)
        {
            if(!st[j] && (t == -1 || dist[j] < dist[t]))
            {
                t = j;
            }
        }
        st[t] = true;
        for(int j = 1;j <= n + m;j ++)
            dist[j] = min(dist[j],dist[t] + g[t][j]);
    }
    int sum = 0,mindist = INF;
    for(int i = 1;i <= n;i ++)
    {
        if(dist[i] > D) return false;
        sum += dist[i];
        mindist = min(mindist,dist[i]);
    }
    if(mindist > maxD)
    {
        maxD = mindist;
        sumD = sum;
        minID = S;
    }
    else if(mindist == maxD)
    {
        if(sum < sumD)
        {
            sumD = sum;
            minID = S;
        }
        else if(sum == sumD)
        {
            if(minID > S) minID = S;
        }
    }
    return true;
}
int main()
{
    memset(g,0x3f,sizeof g);    
    cin >> n >> m >> k >> D;
    string a,b;
    int w;
    while(k --)
    {
        cin >> a >> b >> w;
        int x = get_num(a),y = get_num(b);
        //cout << x << ' ' << y << ' ' << w << endl;
        g[x][y] = g[y][x] = min(g[x][y],w);
    }
    int maxD = 0,sumD = INF,minID = INF;
    for(int i = n + 1;i <= n + m;i ++)
    {
        bool is_valid = dijkstra(i,maxD,sumD,minID);//本来我的设计是想让函数返回很多值的，一个距离，一个总和，一个是否合法
        if(!is_valid) continue;//然后发现其实“可以把多个参数传引用进去”，以达到同样的效果。
    }
    if(minID != INF)
    {
        //cout << minID << ' ' << maxD << ' ' << sumD << endl;
        printf("G%d\n",minID - n);
        printf("%.1lf %.1lf",(double)maxD,(double)sumD / n + 1e-8);
    }
    else printf("No Solution");
    return 0;
}

紧急情况
2022/11/15

题目链接：https://www.acwing.com/problem/content/1477/

作为城市的紧急救援团队负责人，你将获得一张你所在国家的特殊地图。

该地图显示了一些通过道路连接的分散城市，道路是双向的。

地图上标出了每个城市的救援队数量以及每对城市之间的每条道路的长度。

当其他城市发出紧急求援信息时，你的工作是尽快带领你的士兵前往该地点，同时，在途中尽可能多地调动救援帮手。

输入格式

第一行包含四个整数 $N$，表示城市数量（城市编号从 $0$ 到 $N-1$），$M$ 表示道路数量，$C_1$ 表示你当前所在的城市编号，$C_2$ 表示发出紧急求援信息的城市编号。

第二行包含 $N$ 个整数，其中第 $i$ 个整数表示城市 $i$ 的救援队数量。

接下来 $M$ 行，每行包含三个整数 $c_1,c_2,L_i$，表示城市 $c_1$ 和城市 $c_2$ 之间存在一条道路相连，道路长度为 $L_i$。

数据保证 $C_1$ 和 $C_2$ 之间至少存在一条路径相连。

输出格式

共一行，两个整数，第一个整数表示 $C_1$ 和 $C_2$ 之间最短路的数量，第二个整数表示走最短路的情况下，能聚集到的救援队最大数量。

数据范围

$2 \le N \le 500$,
$1 \le M \le 600$,
$1 \le L_i \le 200$,
每个城市包含的救援人员数量不超过 $200$。

输入样例：

5 6 0 2
1 2 1 5 3
0 1 1
0 2 2
0 3 1
1 2 1
2 4 1
3 4 1

输出样例：

2 4


对于一个点，可能会被多次更新

#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;
const int N = 610;
int n,m,S,T;
int g[N][N],w[N];
int dist[N],cnt[N],sum[N];
bool st[N];
void dijkstra()
{
    memset(dist,0x3f,sizeof dist);
    dist[S] = 0,cnt[S] = 1,sum[S] = w[S];
    for(int i = 0;i < n;i ++)
    {
        int t = -1;
        for(int j = 0;j < n; j ++)
        {
            if(!st[j] && (t == -1 || dist[j] < dist[t]))
            {
                t = j;
            }
        }
        //cout << t << endl;
        st[t] = true;
        for(int j = 0;j < n; j ++)
        {
            if(dist[j] > dist[t] + g[t][j])//当最短路径被更新的时候
            {
                dist[j] = dist[t] + g[t][j];
                cnt[j] = cnt[t];
                sum[j] = sum[t] + w[j];
            }
            else if(dist[j] == dist[t] + g[t][j])//如果有等长的最短路径
            {
                cnt[j] += cnt[t];
                sum[j] = max(sum[j],sum[t] + w[j]);
            }
        }
    }
}
int main()
{
    memset(g,0x3f,sizeof g);
    cin >> n >> m >> S >> T;
    for(int i = 0;i < n; i ++) cin >> w[i];
    while(m --)
    {
        int a,b,c;
        cin >> a >> b >> c;
        g[a][b] = g[b][a] = min(g[a][b],c);
    }
    dijkstra();
    printf("%d %d",cnt[T],sum[T]);
    return 0;
}

ICS-1-C语言机制
2022/4/14

C程序执行的两个视角

静态: C代码的连续一段 总能对应到 一段连续的机器指令
动态： C代码执行的状态总能对应到 机器的状态
●源代码视角 

◾函数、变量、指针...

●机器指令视角

◾ 寄存器、内存、地址...

两个视角共同之处： 内存
◾ 代码、变量（源代码视角） = 地址 + 长度（机器指令视角）
◾ 内存 = 代码 + 数据 + 堆栈
因此理解C程序执行最重要的就是内存模型

C语言中一切皆是地址
指针都是纯粹的，类型声明只不过是对一段内存的解读方式 

#include <stdio.h>
void printptr(void *p)//空指针是非常纯粹的，它就是一个纯粹的地址，强制类型转换更改了内存的解读方式
{
    printf("*p = %p       *p = %016lx\n",p,*(long *)p);//对于p来说，首先它是一个指针，剩下的那个long代表对指针指向的内存的一个解读
}
int x;
int main(int argc,char *argv[])
{
    printptr(main);//代码
    printptr(&main);
    printptr(&x);//数据
    printptr(&argc);//堆栈
    printptr(argv);
    printptr(&argv);
    printptr(argv[0]);
}

        */
        f(argc - 1,argv + 1);
    }
}

C语言机制使得代码可读并且可维护
2022/4/14


1、软件里有很多隐藏的dependence（依赖关系），有一些额外的，代码中没有体现和约束的“规则”
2、往往如果增加一个需求，就需要改动很多地方
一个例子是如果要模拟一个寄存器的工作，现在要加一个新的寄存器，就需要更改：X定义的位置、X1定义的位置、计算模拟的位置、写回的位置。如果这四个位置分布在一篇代码里还好，如果每个位置都分布在不同的文件中那就麻烦了

● 一处改了，另外一处就忘了(例如数组中加了一个寄存器忘记更新NREG...)
● 减少依赖关系 -> 降低代码耦合程度


好处在于只需要定义一次列表就可以对列表进行不同的处理，这种形式很利于维护
不至于把同一段代码copy很多次，很大程度上促进了代码的复用

#include <stdint.h>
#define NREG 4

typedef uint8_t u8;//定义一个8bit的int
u8 pc = 0,R[NREG];
/*
小知识：
还存在一个intptr_t这种类型，保证了int里面能够容纳指针
对于(int)ptr这种操作是非常危险的，如果在32位机里面ptr是4字节，int也是4字节那还好
但是如果把同样一段代码移植到了64位机里面int是4字节，而ptr是8字节，此时如果将指针cast成int的话那就JJ
*/

int main()
{


    //方法1
    u8 R[] = {
        0,0,0,0,0......init_pc,
    };//这种定义不好维护，很容易就漏写0，而且如果需求有更改（要再多加一个寄存器）比较麻烦


    //方法2
    enum {RA,R1 ..... PC};
    u8 R[] = {
        [RA] = 0,
        [R1] = 0,
        ...
        [PC] = init_pc,
    }//这种定义的手法，给予了代码很好的可读性，使得代码“不言自明”，而且如果要多加一个寄存器的话也只需要改列表，然后多加一个就好了

    return 0;
}
