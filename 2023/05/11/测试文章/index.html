矩阵乘法 + 快速幂求解经典斐波那契数列

2023/5/11 数学

$$\begin{equation}
H_x=\frac{1}{3}\times{
\left[ \begin{array}{ccc}
1 & 0 & -1\
1 & 0 & -1\
1 & 0 & -1
\end{array}
\right ]},
H_y=\frac{1}{3}\times{
\left[ \begin{array}{ccc}
1 & 1 & 1\
0& 0 & 0\
1 & 1 & 1
\end{array}
\right ]}
\end{equation}$$

题目链接：https://www.acwing.com/solution/content/47987/

#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;
const int N = 3;
typedef long long LL;
int n,m;
void mul(int c[][N],int a[][N],int b[][N])//这里直接是传引用的形式进来
{
    static int temp[N][N];
    memset(temp,0,sizeof temp);
    for (int i = 0;i < N;i ++)
        for (int j = 0;j < N;j ++)
            for (int k = 0;k < N;k ++)
                temp[i][j] = (temp[i][j] + (LL)a[i][k] * b[k][j]) % m;

    memcpy(c,temp,sizeof temp);        
}
int main()
{
    cin >> n >> m;
    /*f1[3][3] 是一个矩阵；写成二维使得我们只需要写一个乘法
    1 1 1
    0 0 0
    0 0 0
    */
    int f1[3][3] = {0,1,0};
    int A[3][3] = {
        {0,1,0},
        {1,1,1},
        {0,0,1}
    };

    for (int i = n;i;i >>= 1)
    {
        if (i & 1) mul(f1,f1,A);//F1 = F1 * A;
        mul(A,A,A);//A = A * A;
    }
    cout << f1[0][2];
    return 0;
}
