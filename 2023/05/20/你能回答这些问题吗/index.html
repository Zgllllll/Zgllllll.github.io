你能回答这些问题吗

2023/5/20 算法 数据结构

题目链接：https://www.acwing.com/problem/content/246/

给定长度为 $N$ 的数列 $A$，以及 $M$ 条指令，每条指令可能是以下两种之一：

  1. 1 x y，查询区间 $[x,y]$ 中的最大连续子段和，即 $\max\limits_{x \le l \le r \le y}${$\sum\limits^r_{i=l} A[i]$}。
  2. 2 x y，把 $A[x]$ 改成 $y$。

对于每个查询指令，输出一个整数表示答案。

输入格式

第一行两个整数 $N,M$。

第二行 $N$ 个整数 $A[i]$。

接下来 $M$ 行每行 $3$ 个整数 $k,x,y$，$k=1$ 表示查询（此时如果 $x>y$，请交换 $x,y$），$k=2$ 表示修改。

输出格式

对于每个查询指令输出一个整数表示答案。

每个答案占一行。

数据范围

$N \le 500000, M \le 100000$,
$-1000 \le A[i] \le 1000$

输入样例：

5 3
1 2 -3 4 5
1 2 3
2 2 -1
1 3 2

输出样例：

2
-1

最值区间和线段树

#include <iostream>
#include <algorithm>
using namespace std;
const int N = 5e5 + 10,M = 1e5 + 10;
int n,m;
int w[N];
struct Node
{
  int l,r;
  int sum,lmax,rmax,tmax;
}tr[N * 4];
void pushup(Node& u, Node& l, Node& r)
{
    u.sum = l.sum + r.sum;
    u.lmax = max(l.lmax, l.sum + r.lmax);
    u.rmax = max(r.rmax, r.sum + l.rmax);
    u.tmax = max({l.tmax, r.tmax, l.rmax + r.lmax});
}
void pushup(int u)
{
    pushup(tr[u], tr[u << 1], tr[u << 1 | 1]);//这个封装的很妙。pushup操作我们需要三个对象，一个父节点，三个子节点。
}
void build(int u,int l,int r)
{
    tr[u].l = l,tr[u].r = r;//这个可以先标记上。而剩下的信息，如sum,lmax...等，需要子节点的返回
    if (l == r) 
    {
        tr[u] = {l,r,w[l],w[l],w[l],w[l]};
        return;
    }
    else
    {
        int mid = l + r >> 1;
        build(u << 1, l, mid),build(u << 1 | 1, mid + 1, r);
        pushup(u);
    }
}
void change(int u,int x,int v)
{
    if (tr[u].l == x && tr[u].r == x) 
    {
        tr[u] = {x, x, v, v, v, v};
        return;
    }
    else
    {
        int mid = tr[u].l + tr[u].r >> 1;
        if (x <= mid) change(u << 1, x, v);
        else if (x > mid) change(u << 1 | 1, x, v);
        pushup(u);
    }
}
//本题的查询，和上题目的查询不大一样
Node query(int u, int l, int r)
{
    if (tr[u].l >= l && tr[u].r <= r) return tr[u];//当前点完全被查询的大线段包含（这是一种上面大，下面小的状态）（默认上面的是要查的，下面的是树的节点）
    else//在下面的节点不被上面节点包含情况下(下面的节点左端点或右端点比起上面的肯定有出头)
    {
        int mid = tr[u].l + tr[u].r >> 1;
        if (r <= mid) return query(u << 1, l, r);//如果只涉及下面结点的左半边。即下面的节点，右半部分有出头
        else if (l > mid) return query(u << 1 | 1, l, r);//如果下面的节点，左半部分有出头
        else//l <= mid 或 r > mid，和第一题的那个查询一样。需要左右对比看。下面的两节点都需要返回信息
        {
            auto left = query(u << 1, l, r);
            auto right = query(u << 1 | 1, l, r);
            Node res;
            pushup(res, left, right);
            return res;
        }
    }
}


int main()
{
    cin >> n >> m;
    for (int i = 1;i <= n;i ++) cin >> w[i];
    build(1, 1, n);
    int k,x,y;
    while (m --)
    {
        cin >> k >> x >> y;
        if (k == 1)
        {
            if (x > y) swap(x,y);
            Node t = query(1, x, y);
            cout << t.tmax << endl;
        }
        else
        {
            change(1, x, y);
        }
    }
    return 0;
}```
